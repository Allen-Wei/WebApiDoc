 public class ResultMessage
 {
     public int Code { get; set; }
     public float DetailCode { get; set; }
     public bool Success { get; set; }
     public string CodeBrief { get; set; }
     public string Brief { get; set; }

 }
返回结果JSON格式
{"Code":200,"DetailCode":0.0,"Success":true,"CodeBrief":"ok","Brief":"","Message":"","Result":""}
描述
Code: 状态码.
DetailCode: 详细状态码.
Success: 操作是否成功.
CodeBrief: 状态码简述.
Brief: 返回结果简述.
Message: 返回结果详述.
Result: 返回结果
